function esMercurioRetrógrado(mercurioActual, mercurioAnterior) {
  // Obtener la longitud eclíptica actual y anterior de Mercurio
  const longitudActual = mercurioActual.raw.longitude;
  const longitudAnterior = mercurioAnterior.raw.longitude;

  // Si la longitud actual es menor que la anterior, Mercurio está retrógrado
  return longitudActual < longitudAnterior;
}

// Objeto de Mercurio en el momento actual y anterior
const mercurioActual = {
  raw: {
     // longitude: 4.2036004075498425
     longitude:   5.801558382437137
  }
};

const mercurioAnterior = {
  raw: {
      //longitude: 5.665827134322659
      longitude:   4.053695893690032
  }
};

// Verificar si Mercurio está retrógrado en este momento
const estáRetrógrado = esMercurioRetrógrado(mercurioActual, mercurioAnterior);
console.log('¿Mercurio está retrógrado?', estáRetrógrado ? 'Sí' : 'No');
