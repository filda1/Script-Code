async function connectToDatabase() {
  try {
    const connection = await createConnection({
      type: 'mssql',

      port: 1433,

      database: '',
      entities: [Table],

    const scopes: Watt_Scopes[] = await connection.manager.find(Watt_Scopes);

      console.log(`Name: ${scope.name}`);

    console.log('scopes :::::::::>', scopes)
